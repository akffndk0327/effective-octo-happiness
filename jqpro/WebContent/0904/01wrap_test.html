wrap() : 각 줄에 div가 생김 .
-$('p').wrap('<div>'); p요소를 각각 div요소로 감싸기 

warpAll() : p요소 전체를 하나의 div요소로 감쌈
-$('p').wrapAll('<div>');

$('p').on('click',function(){
  $(this).wrap('<div/>');
 	})
